bootstrap js 적용에 필요한 설정

bootstrap.bundle.min.js는 aync 적용시에 head에서 하면안되고 body에 적용해야함 

   async function loadHeadLinks(){
      let links = await fetch('/utils/links.html');
      let linksText = await links.text();
      여기에서 기존 document.head하면 안되고 body 끝에 삽입해야함
      document.body.insertAdjacentHTML("afterbegin",linksText);
   }
